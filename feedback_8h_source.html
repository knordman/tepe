25 real_t hinge_angle(struct mem_t *m, index_t hinge_num)
26 {
27  index_t b0 = _Jm(m, 5*hinge_num, 0);
28  index_t b1 = _Jm(m, 5*hinge_num, 1);
29 
30  // Relative rotation between b0 and b1
31  tp_quatern q0;
32  get_quatern(quatern(m, b0), q0);
33 
34  tp_quatern q1;
35  get_quatern(quatern(m, b1), q1);
36 
37  q0[1] *= TP_REAL(-1.0); q0[2] *= TP_REAL(-1.0); q0[3] *= TP_REAL(-1.0);
38 
39  tp_quatern dq;
40  mult_quatern_quatern(dq, q0, q1);
41 
42  tp_quatern _iniquatern;
43  get_quatern(iniquatern(m, hinge_num), _iniquatern);
44  _iniquatern[1] *= TP_REAL(-1.0); _iniquatern[2] *= TP_REAL(-1.0); _iniquatern[3] *= TP_REAL(-1.0);
45 
46  tp_quatern hdq;
47  mult_quatern_quatern(hdq, dq, _iniquatern);
48 
49  // Convert quaternion to angle (from ODE source)
50  tp_vec3 axis;
51  get_vec3(haxis(m, hinge_num), axis);
52  tp_mtx33 R0;
53  get_mtx33(R(m, b0), R0);
54  mult_to_mtx33_vec3(R0, axis);
55 
56  real_t cost2 = hdq[0];
57  real_t sint2 = sqrt(hdq[1]*hdq[1] + hdq[2]*hdq[2] + hdq[3]*hdq[3]);
58 
59  real_t theta = (dot_vec3(hdq + 1, axis) >= 0) ?
60  (TP_REAL(2.0) * atan2(sint2, cost2) ) :
61  (TP_REAL(2.0) * atan2(sint2, -cost2) );
62 
63  theta -= (theta > TP_PI)*(TP_REAL(2.0) * TP_PI);
64 
65  return theta;
66 }
