13 struct mem_t
14 {
15  real_t q[(TP_BODIES)*(TP_SIZE_VEC3+TP_SIZE_VEC4)]; // Generalized position variable, pos + quatern
16  real_t v[(TP_BODIES)*TP_SIZE_VEC6]; // Generalized velocity variable, vel + omega
17  real_t mi[(TP_BODIES)]; // Inverse mass
18  real_t Ibi[(TP_BODIES)*3*TP_SIZE_VEC3]; // Inverse inertia matrix
19  real_t R[(TP_BODIES)*3*TP_SIZE_VEC3]; // Convenience matrix
20 
21  real_t Fe[(TP_BODIES)*TP_SIZE_VEC6]; // External force
22 
23  real_t J[2*TP_SIZE_VEC6*TP_CONSTRAINTS]; // Constraint Jacobian
24 
25  real_t lambda[TP_CONSTRAINTS]; // F_c = J^{T}\lambda
26  real_t lambda_min[TP_CONSTRAINTS]; // min
27  real_t lambda_max[TP_CONSTRAINTS]; // max
28 
29  index_t mm[(TP_MOTORS)]; // Mapping motors->hinges
30  real_t mdspeed[(TP_MOTORS)]; // Desired speed for motors
31  real_t iniq[(TP_HINGES)*TP_SIZE_VEC4]; // Quaternions for initial rotations
32 
33  index_t Jm[2*TP_CONSTRAINTS]; // Mapping->bodies, sparse Jacobian
34  real_t B[2*TP_SIZE_VEC6*TP_CONSTRAINTS]; // M^{-1}J^{T}, for solving
35  real_t a[(TP_BODIES)*TP_SIZE_VEC6]; // B\lambda, for solving
36  real_t d[TP_CONSTRAINTS]; // diag(JB), for solving
37  real_t rhs[TP_CONSTRAINTS]; // Right hand side, for solving
38 
39  real_t haxes[(TP_HINGES)*2*TP_SIZE_VEC6]; // Hinge axis 1+2, tangent base 1
40  real_t hanchors[(TP_HINGES)*TP_SIZE_VEC6]; // Hinge anchors ( -''- )
41 
42 #ifdef TP_DEBUG
43  real_t Fc[(TP_BODIES)*TP_SIZE_VEC6]; // Constraint force
44  real_t cinfo[3*(TP_FEET)*TP_SIZE_VEC6]; // Contact points + contact normals
45  real_t cplane[(TP_FEET)*TP_SIZE_VEC6]; // Contact plane (axes where slip is eliminated)
46  index_t cbody[(TP_FEET)]; // Body indexes connected to feet
47 #endif
48 };

51 void zero_memory(struct mem_t *mem)
52 {
53  for(size_t i = 0; i < (TP_BODIES)*(TP_SIZE_VEC3+TP_SIZE_VEC4); ++i) mem->q[i] = TP_REAL(0.0);
54  for(size_t i = 0; i < (TP_BODIES)*TP_SIZE_VEC6; ++i) mem->v[i] = TP_REAL(0.0);
55  for(size_t i = 0; i < (TP_BODIES); ++i) mem->mi[i] = TP_REAL(0.0);
56  for(size_t i = 0; i < (TP_BODIES)*3*TP_SIZE_VEC3; ++i) mem->Ibi[i] = TP_REAL(0.0);
57  for(size_t i = 0; i < (TP_BODIES)*3*TP_SIZE_VEC3; ++i) mem->R[i] = TP_REAL(0.0);
58  for(size_t i = 0; i < (TP_BODIES)*TP_SIZE_VEC6; ++i) mem->Fe[i] = TP_REAL(0.0);
59  for(size_t i = 0; i < 2*TP_SIZE_VEC6*TP_CONSTRAINTS; ++i) mem->J[i] = TP_REAL(0.0);
60 
61  for(size_t i = 0; i < TP_CONSTRAINTS; ++i) mem->lambda[i] = TP_REAL(0.0);
62  for(size_t i = 0; i < TP_CONSTRAINTS; ++i) mem->lambda_min[i] = TP_REAL(-1048576.0);
63  for(size_t i = 0; i < TP_CONSTRAINTS; ++i) mem->lambda_max[i] = TP_REAL(1048576.0);
64 
65  for(size_t i = 0; i < 2*TP_CONSTRAINTS; ++i) mem->Jm[i] = 0;
66  for(size_t i = 0; i < 2*TP_SIZE_VEC6*TP_CONSTRAINTS; ++i) mem->B[i] = TP_REAL(0.0);
67  for(size_t i = 0; i < (TP_BODIES)*TP_SIZE_VEC6; ++i) mem->a[i] = TP_REAL(0.0);
68  for(size_t i = 0; i < TP_CONSTRAINTS; ++i) mem->d[i] = TP_REAL(0.0);
69  for(size_t i = 0; i < TP_CONSTRAINTS; ++i) mem->rhs[i] = TP_REAL(0.0);
70  for(size_t i = 0; i < (TP_HINGES)*TP_SIZE_VEC6; ++i) mem->hanchors[i] = TP_REAL(0.0);
71  for(size_t i = 0; i < (TP_HINGES)*2*TP_SIZE_VEC6; ++i) mem->haxes[i] = TP_REAL(0.0);
72 
73  for(size_t i = 0; i < (TP_MOTORS); ++i) mem->mm[i] = 0;
74  for(size_t i = 0; i < (TP_MOTORS); ++i) mem->mdspeed[i] = TP_REAL(0.0);
75  for(size_t i = 0; i < (TP_HINGES)*TP_SIZE_VEC4; ++i) mem->iniq[i] = TP_REAL(0.0);
76 
77 #ifdef TP_DEBUG
78  for(size_t i = 0; i < (TP_BODIES)*TP_SIZE_VEC6; ++i) mem->Fc[i] = TP_REAL(0.0);
79  for(size_t i = 0; i < 3*(TP_FEET)*TP_SIZE_VEC6; ++i) mem->cinfo[i] = TP_REAL(0.0);
80  for(size_t i = 0; i < (TP_FEET)*TP_SIZE_VEC6; ++i) mem->cplane[i] = TP_REAL(0.0);
81  for(size_t i = 0; i < (TP_FEET); ++i) mem->cbody[i] = 0;
82 #endif
83 }
