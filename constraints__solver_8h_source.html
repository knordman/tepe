260 {
261  // rhs = \frac{1}{\Delta t}\epsilon - \frac{1}{\Delta t}JV - JM^{-1}F_e
262  for(int s = 0; s < TP_CONSTRAINTS; ++s)
263  {
264  real_t JV = TP_REAL(0.0), JMiFe = TP_REAL(0.0);
265 
266  index_t stop_at_body = (s < TP_HINGE_MOTOR_CONSTRAINTS) ? 0 : 1;
267 
268  for(int bi = 1; bi >= stop_at_body; --bi)
269  {
270  index_t body = _Jm(m, s, bi);
271 
272  tp_vec3 _tJ;
273  get_vec3(tJ(m, s, bi), _tJ);
274 
275  tp_vec3 _vel;
276  get_vec3(vel(m, body), _vel);
277 
278  tp_vec3 _aJ;
279  get_vec3(aJ(m, s, bi), _aJ);
280 
281  tp_vec3 _omega;
282  get_vec3(omega(m, body), _omega);
283 
284  // JV (translational) + (rotational)
285  JV += dot_vec3(_tJ, _vel) + dot_vec3(_aJ, _omega);
286 
287  tp_vec3 _tFe;
288  get_vec3(tFe(m, body), _tFe);
289  scale_to_vec3(_tFe, _mi(m, body));
290 
291  tp_vec3 _aFe;
292  get_vec3(aFe(m, body), _aFe);
293 
294  tp_mtx33 _Ibi;
295  get_mtx33(Ibi(m, body), _Ibi);
296 
297  tp_mtx33 _R;
298  get_mtx33(R(m, body), _R);
299 
300  tp_mtx33 IbiRT;
301  mult_mtx33_mtx33T(IbiRT, _Ibi, _R);
302 
303  tp_mtx33 Ii;
304  mult_mtx33_mtx33(Ii, _R, IbiRT);
305 
306  mult_to_mtx33_vec3(Ii, _aFe);
307 
308  // JM^{-1}Fe (translational) + (rotational)
309  JMiFe += dot_vec3(_tJ, _tFe) + dot_vec3(_aJ, _aFe);
310  }
311 
312  *rhs(m, s) = -(TP_REAL(1.0)/dt) * JV - JMiFe;
313  }

437 void solve_for_lambda(struct mem_t *m, real_t dt, int num_iterations)
438 {
439  // Projected Gauss-Seidel, as from Catto paper
440 
441  /* CUDA: Better to compute on empty rows, or better to quit
442  * and wait in a synchronization? Test? For now, empty rows!
443  */
444 
445  // Solves JB\lambda = rhs (J = sparse, B = sparse)
446  compute_B(m); // B = M^{-1}J^{T}
447  compute_a(m); // a = B\lambda_0
448  compute_d(m); // d = diag(JB)
449  compute_rhs(m, dt); // rhs = 1/dt*e - 1/dt*Jv - M^{-1}F_e
450 
451  for(int i = 0; i < num_iterations; ++i)
452  {
453 // real_t delta = 0.0;
454  for(int s = 0; s < TP_CONSTRAINTS; ++s)
455  {
456  index_t stop_at_body = (s < TP_HINGE_MOTOR_CONSTRAINTS) ? 0 : 1;
457 
458  real_t tmp = TP_REAL(0.0);
459  for(int bi = 1; bi >= stop_at_body; --bi)
460  {
461  index_t body = _Jm(m, s, bi);
462 
463  tp_vec3 _tJ;
464  get_vec3(tJ(m, s, bi), _tJ);
465 
466  tp_vec3 _ta;
467  get_vec3(ta(m, body), _ta);
468 
469  tmp += dot_vec3(_tJ, _ta);
470 
471  tp_vec3 _aJ;
472  get_vec3(aJ(m, s, bi), _aJ);
473 
474  tp_vec3 _aa;
475  get_vec3(aa(m, body), _aa);
476 
477  tmp += dot_vec3(_aJ, _aa);
478  }
479 
480  // Fix to avoid d = 0
481 // real_t dfix = _d(m, s) + (abs(_d(m, s) < 1e-7))*1e7;
482 // real_t delta_lambda = (_rhs(m, s) - tmp) / dfix;
483 
484  real_t delta_lambda = TP_REAL(0.0);
485  if(_d(m, s) > TP_REAL(1e-7) || _d(m, s) < TP_REAL(-1e-7))
486  delta_lambda = (_rhs(m, s) - tmp) / _d(m, s);
487 
488 // if(s == 40)
489 // delta = delta_lambda;
490  // Limit lambda
491  real_t new_lambda = clamp2(_lambda(m, s), delta_lambda, _lambda_min(m, s), _lambda_max(m, s));
492 
493  delta_lambda = new_lambda - _lambda(m, s);
494 
495  *lambda(m, s) += delta_lambda;
496 
497  for(int bi = 1; bi >= stop_at_body; --bi)
498  {
499  index_t body = _Jm(m, s, bi);
500 
501  *x(ta(m, body)) += delta_lambda * _x(tB(m, s, bi));
502  *y(ta(m, body)) += delta_lambda * _y(tB(m, s, bi));
503  *z(ta(m, body)) += delta_lambda * _z(tB(m, s, bi));
504 
505  *x(aa(m, body)) += delta_lambda * _x(aB(m, s, bi));
506  *y(aa(m, body)) += delta_lambda * _y(aB(m, s, bi));
507  *z(aa(m, body)) += delta_lambda * _z(aB(m, s, bi));
508  }
509  }
